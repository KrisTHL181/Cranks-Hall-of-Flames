\mathcal{T} = \begin{pmatrix} 0 & 1 \\ 1 & 0 \end{pmatrix},

\mathcal{T}|0_L\rangle = |1_L\rangle, \quad \mathcal{T}|1_L\rangle = |0_L\rangle.
