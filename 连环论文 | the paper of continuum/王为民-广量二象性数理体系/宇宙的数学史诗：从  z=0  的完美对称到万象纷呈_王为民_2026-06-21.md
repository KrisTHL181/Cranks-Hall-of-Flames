\frac{d^2 z}{dt^2} + 3H\frac{dz}{dt} + \frac{\partial V_W}{\partial z^\dagger} = 0, \quad H^2 = \frac{8\pi G}{3}\left[\left|\frac{d\mathcal{T}}{dt}\right|^2 + V_W(z)\right]
