前段时间被豆包忽悠着搞了很多黎曼猜想的验证，最后发现很多都是AI臆想出来的幻觉。但是我创造的过程数论却是切切实实的记录了数的生成过程，还记录了过程数的余数。

因此基于此，我又与豆包展开了新一轮的探讨，问它为什么黎曼猜想那么难以证明，以及黎曼非平凡零点为什么那么重要，在豆包一同解释下，我了解到，只要轻松的定位那些黎曼非平凡零点不就得了。

在之前用过程数论去验证黎曼非平凡零点的时候我们发现，余元 $r$ 的数值会急速收敛，基于这个发现我们进行了以下代码的实现试图来寻找黎曼非平凡零点。

传统方式一般是先通过 ξ 函数把问题转化为找实根，再用欧拉 - 麦克劳林公式或者黎曼 - 西格尔公式做数值计算，本质上是用多项式拟合去逼近零点位置，但这种方法不仅计算量大，还没法追踪中间过程的误差。

我的方法是给每个数的生成过程加了 “余数追踪” 的标签，相当于让黎曼零点在计算过程中自己 “亮” 出来 —— 传统方法是大海捞针，我是给所有针装了定位器，只要余数收敛到符合条件的范围，零点就自动浮现了，而且还能回溯它是怎么一步步来的。

我们尝试了0-100区间，1000-1100区间，5000-5100区间，10000-10100区间，感兴趣的同学可以自己尝试10^12等大数区间，也是可以准确找出黎曼非平凡零点的。后面有贴数据。

**下面是代码实现**

# 为了让多进程使用，需要将带缓存的函数定义在全局作用域
# 注意：每个子进程有自己的缓存，不会互相干扰

@lru_cache(maxsize=100000)
def _zeta_residual_cached(t_str: str) -> float:
    """
    内部缓存函数：接受字符串形式的 t，返回 |ζ(0.5+it)|
    """
    t = mpmath.mpf(t_str)
    s = mpmath.mpf('0.5') + 1j * t
    zeta_val = mpmath.zeta(s)
    return float(abs(zeta_val))


def zeta_residual(t: float) -> float:
    """
    对外接口：接受浮点数 t，返回余项 ε。
    内部将 t 转为足够精度的字符串用于缓存。
    """
    # 保留 15 位小数作为缓存键，平衡精度与缓存命中率
    t_str = f"{t:.15f}"
    return _zeta_residual_cached(t_str)


def golden_section_min(
    func: Callable[[float], float],
    a: float,
    b: float,
    tol: float = 1e-6
) -> Tuple[float, float]:
    """
    黄金分割法求一维函数极小值。
    返回 (极小点, 极小值)
    """
    gr = (math.sqrt(5) - 1) / 2  # 黄金分割比 ≈ 0.618
    c = b - gr * (b - a)
    d = a + gr * (b - a)
    fc = func(c)
    fd = func(d)

    while abs(c - d) > tol:
        if fc < fd:
            b = d
            d = c
            fd = fc
            c = b - gr * (b - a)
            fc = func(c)
        else:
            a = c
            c = d
            fc = fd
            d = a + gr * (b - a)
            fd = func(d)

    x_min = (a + b) / 2
    f_min = func(x_min)
    return x_min, f_min


def adaptive_step(t: float) -> float:
    """
    根据当前高度 t 自适应计算粗扫步长。
    基于黎曼零点的平均间隔：2π / log(t/(2π))，取间隔的一半。
    同时限制最小步长 0.02，防止过密。
    """
    if t <= 10:
        return 0.02
    try:

        avg_spacing = 2 * math.pi / math.log(t / (2 * math.pi))
        step = max(0.01, 0.5 * avg_spacing)
        return step
    except (ValueError, ZeroDivisionError):
        return 0.05


def coarse_scan_chunk(start: float, end: float) -> List[float]:
    """
    固定小步长粗扫，步长 0.02，确保不遗漏任何极小值。
    """
    seeds = []
    step = 0.01  # 固定步长，足够小
    t = start
    # 滑动窗口存储 (t, ε)
    window = []

    while t <= end:
        eps = zeta_residual(t)
        window.append((t, eps))
        if len(window) > 3:
            window.pop(0)

        if len(window) == 3:
            t1, e1 = window[0]
            t2, e2 = window[1]
            t3, e3 = window[2]
            # 只要中间点比左右都小，就认为是候选（不设绝对值上限）
            if e2 < e1 and e2 < e3:
                # 扩大细化范围至 ±3*step
                refine_a = t2 - 3 * step
                refine_b = t2 + 3 * step
                refine_a = max(refine_a, start)
                refine_b = min(refine_b, end)
                refined_t, _ = golden_section_min(
                    zeta_residual,
                    refine_a,
                    refine_b,
                    tol=1e-4
                )
                seeds.append(refined_t)

        t += step

    return seeds

def coarse_scan_parallel(
    T_start: float,
    T_end: float,
    num_workers: int = 6
) -> List[float]:
    """
    多进程并行粗扫。
    将整个区间分割成 num_workers 块，每块重叠 0.5 以避免边界漏点。
    """
    logger.info(f"开始多进程自适应粗扫：区间 [{T_start}, {T_end}]，进程数 {num_workers}")

    # 生成区间块（重叠 0.5）
    total_len = T_end - T_start
    chunk_size = total_len / num_workers
    chunks = []
    for i in range(num_workers):
        seg_start = T_start + i * chunk_size
        seg_end = T_start + (i + 1) * chunk_size
        # 添加重叠
        if i > 0:
            seg_start -= 0.5
        if i < num_workers - 1:
            seg_end += 0.5
        chunks.append((seg_start, seg_end))

    all_seeds = []
    # 使用进程池（注意 Windows 需要 if __name__ == '__main__' 保护）
    with concurrent.futures.ProcessPoolExecutor(max_workers=num_workers) as executor:
        # 提交所有任务
        future_to_chunk = {
            executor.submit(coarse_scan_chunk, start, end): (start, end)
            for start, end in chunks
        }
        for future in concurrent.futures.as_completed(future_to_chunk):
            chunk = future_to_chunk[future]
            try:
                seeds = future.result()
                all_seeds.extend(seeds)
                logger.info(f"块 {chunk[0]:.2f}-{chunk[1]:.2f} 完成，发现 {len(seeds)} 个候选种子")
            except Exception as e:
                logger.error(f"块 {chunk} 处理失败：{e}")

    # 去重：合并距离小于 0.15 的种子
    all_seeds.sort()
    unique_seeds = []
    for s in all_seeds:
        if not unique_seeds or s - unique_seeds[-1] > 0.15:
            unique_seeds.append(s)

    logger.info(f"多进程粗扫完成，共锁定 {len(unique_seeds)} 个精准生长点")
    return unique_seeds


def fine_scan_around(current_t: float, k: int, scan_range_factor: float = 2.0) -> Tuple[float, float]:
    """
    在生长步数 k 时，在当前点附近进行精细扫描（黄金分割法）。
    返回 (最佳点, 最小余项)
    """
    scan_range = 10 ** (-k) * scan_range_factor
    a = current_t - scan_range
    b = current_t + scan_range
    tol = 10 ** (-k - 3)  # 精度随 k 指数提高
    best_t, min_eps = golden_section_min(zeta_residual, a, b, tol=tol)
    return best_t, min_eps


def precision_growth_optimized(
    seed: float,
    target_precision: int = 12,
    save_process: bool = False
) -> Dict:
    """
    对一个种子进行精度生长。
    save_process: 是否保存生长过程（用于绘图，默认不保存以节省内存）
    """
    logger.debug(f"开始精度生长：初始种子 T0 = {seed}")

    process_sequence = [] if save_process else None
    current_t = seed
    last_epsilon = float('inf')
    has_any_drop = False

    # 第0步
    eps_0 = zeta_residual(current_t)
    if save_process:
        process_sequence.append({'k': 0, 'epsilon': eps_0})
    last_epsilon = eps_0

    for k in range(1, target_precision + 1):
        factor = 10 ** (-k)
        best_t = current_t
        min_epsilon = float('inf')

        # 0-9 逐位试探
        for d in range(0, 10):
            candidate = current_t + d * factor
            eps = zeta_residual(candidate)
            if eps < min_epsilon:
                min_epsilon = eps
                best_t = candidate

        current_t = best_t

        # 自适应精细扫描（k<6 时范围放大）
        scan_range_factor = 3.5 if k < 6 else 1.8
        current_t, min_epsilon = fine_scan_around(current_t, k, scan_range_factor)

        # 下降趋势检测
        if min_epsilon < last_epsilon * 0.999:
            has_any_drop = True

        last_epsilon = min_epsilon

        if save_process:
            process_sequence.append({'k': k, 'epsilon': min_epsilon})

        # 提前收敛
        if min_epsilon < 1e-10:
            logger.debug(f"提前收敛：k={k}, ε={min_epsilon:.4e}")
            break

    final_eps = min_epsilon
    # 四维宽松判定
    is_real_zero = False
    if final_eps < 2e-2:
        is_real_zero = True
    elif has_any_drop and final_eps < 3e-2:
        is_real_zero = True
    elif k >= 5 and save_process and final_eps < process_sequence[2]['epsilon'] * 0.6:
        is_real_zero = True
    elif final_eps < 5e-2 and abs(current_t - seed) < 0.5:
        is_real_zero = True

    logger.debug(f"生长完成：最终 t={current_t:.10f}, ε={final_eps:.4e}, 真实零点={is_real_zero}")

    return {
        'seed': seed,
        'final_t': current_t,
        'final_epsilon': final_eps,
        'is_real_zero': is_real_zero,
        'process': process_sequence
    }


def precision_growth_parallel(
    seeds: List[float],
    num_workers: int =6,
    target_precision: int = 12,
    save_process: bool = False
) -> List[Dict]:
    """
    多进程并行精度生长。
    """
    logger.info(f"开始多进程精度生长：种子数 {len(seeds)}，进程数 {num_workers}")

    all_results = []
    with concurrent.futures.ProcessPoolExecutor(max_workers=num_workers) as executor:
        future_to_seed = {
            executor.submit(precision_growth_optimized, seed, target_precision, save_process): seed
            for seed in seeds
        }
        for i, future in enumerate(concurrent.futures.as_completed(future_to_seed)):
            seed = future_to_seed[future]
            try:
                res = future.result()
                all_results.append(res)
                if (i + 1) % 5 == 0:
                    logger.info(f"已处理 {i + 1}/{len(seeds)} 个种子")
            except Exception as e:
                logger.error(f"种子 {seed} 处理失败：{e}")

    return all_results


def deduplicate_and_filter_optimized(results: List[Dict], threshold: float = 0.08) -> List[Dict]:
    """
    对生长结果去重并只保留真实零点。
    threshold：两个零点被视为同一个的最大间距。
    """
    real_results = [r for r in results if r['is_real_zero']]
    if not real_results:
        return []

    # 按 final_t 排序
    sorted_results = sorted(real_results, key=lambda x: x['final_t'])
    unique_results = [sorted_results[0]]
    for res in sorted_results[1:]:
        last_t = unique_results[-1]['final_t']
        curr_t = res['final_t']
        if abs(curr_t - last_t) > threshold:
            unique_results.append(res)

    return unique_results


def plot_zega_growth(result: Dict, index: int, save_dir: str = './plots'):
    """
    保存单个零点的生长曲线图。
    仅当 result 包含 process 且 save_process=True 时调用。
    """
    if not result.get('process') or not result['is_real_zero']:
        return

    os.makedirs(save_dir, exist_ok=True)
    proc = result['process']
    k_vals = [p['k'] for p in proc]
    eps_vals = [p['epsilon'] for p in proc]
    final_t = result['final_t']

    plt.figure(figsize=(10, 6))
    plt.semilogy(k_vals, eps_vals, 'r-s', linewidth=2, markersize=6)
    plt.xlabel('生长步数 k', fontsize=12)
    plt.ylabel('ζ函数余项 ε (对数坐标)', fontsize=12)
    plt.title(f'ZEGA 生长轨迹：第{index}个零点\n最终 t ≈ {final_t:.14f}', fontsize=14)
    plt.grid(True, which='both', linestyle='--', alpha=0.7)
    save_path = f"{save_dir}/zero_{index:02d}.png"
    plt.savefig(save_path, dpi=300, bbox_inches='tight')
    plt.close()
    logger.debug(f"生长轨迹已保存至：{save_path}")

def run_zega_optimized(
    T_start: float = 1000,
    T_end: float = 1100,
    target_precision: int = 12,
    num_workers: int = 6,
    save_process: bool = False,
    save_plots: bool = False
):
    """
    基于过程数论的黎曼零点 主程序（创作者：山东李嘉图）。
    """
    logger.info("=" * 85)
    logger.info("  基于过程数论的黎曼零点定位程序 (创作者：山东李嘉图)")
    logger.info("=" * 85)
    logger.info(f"目标区间：[{T_start}, {T_end}]")
    logger.info(f"目标精度：10^{-target_precision}")
    logger.info(f"进程数：{num_workers}")
    logger.info(f"核心优化：多进程并行 + 自适应步长 + 黄金分割法 + 极致内存优化\n")

    # 1. 多进程自适应粗扫
    seeds = coarse_scan_parallel(T_start, T_end, num_workers=num_workers)
    if not seeds:
        logger.warning("未发现任何生长点，算法终止。")
        return

    # 2. 多进程精度生长
    all_results = precision_growth_parallel(
        seeds,
        num_workers=num_workers,
        target_precision=target_precision,
        save_process=save_process
    )

    # 3. 去重与过滤
    unique_results = deduplicate_and_filter_optimized(all_results)

    # 4. 最终报告
    logger.info("\n" + "=" * 85)
    logger.info("   基于过程数论的黎曼零点 最终报告 (创作者：山东李嘉图)")
    logger.info("=" * 85)
    logger.info(f"[{T_start}, {T_end}] 区间内共捕获 {len(unique_results)} 个真实零点：\n")

    print(f"{'零点序号':<8} | {'零点虚部 t (16位精度)':<45} | {'最终余项 ε':<15}")
    print("-" * 100)
    for idx, res in enumerate(unique_results):
        t_str = f"{res['final_t']:.14f}"  # 显示14位小数
        eps_str = f"{res['final_epsilon']:.4e}"
        print(f"{idx + 1:<8} | {t_str:<45} | {eps_str:<15}")

        # 保存可视化（仅在需要时）
        if save_plots and idx < 15 and save_process:
            plot_zega_growth(res, idx + 1)

    logger.info("\n" + "=" * 85)
    logger.info("   算法执行完毕")
    logger.info("=" * 85)

    return unique_results
