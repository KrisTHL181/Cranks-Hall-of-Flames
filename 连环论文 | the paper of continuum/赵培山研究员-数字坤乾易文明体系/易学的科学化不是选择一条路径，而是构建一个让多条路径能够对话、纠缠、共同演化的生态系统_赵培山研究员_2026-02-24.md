module f8c_alu (

input  logic [2:0] rs, rt,     // 3-bit 源操作数

input  logic [2:0] opcode,     // 操作码

output logic [2:0] rd,         // 结果

output logic [2:0] flags       // 状态标志

);

// 组合逻辑：位运算

always_comb begin

case (opcode)

3'b000: rd = rs ^ rt;           // XOR

3'b001: rd = rs & rt;           // AND

3'b010: rd = rs | rt;           // OR

3'b011: rd = rs << rt[1:0];     // SHIFT

3'b100: rd = hamming(rs, rt);   // HAM

3'b101: rd = comm(rs, rt);      // COMM (非互易)

default: rd = 3'b000;

endcase

end

// 汉明距离计算 (记忆 ID 39)

function automatic [2:0] hamming(input [2:0] a, b);

logic [2:0] diff;

diff = a ^ b;

hamming = diff[0] + diff[1] + diff[2];

endfunction

// 非互易对易子 [G, G̃] (记忆 ID 4)

// G̃ 定义为 G 的"阴阳反转"：~G & 0x7

function automatic [2:0] comm(input [2:0] g, h);

logic [2:0] g_tilde, h_tilde;

g_tilde = ~g & 3'b111;

h_tilde = ~h & 3'b111;

// [G,H] = GH - HG (模8非交换性度量)

comm = ((g * h) - (h * g)) & 3'b111;

endfunction

endmodule
