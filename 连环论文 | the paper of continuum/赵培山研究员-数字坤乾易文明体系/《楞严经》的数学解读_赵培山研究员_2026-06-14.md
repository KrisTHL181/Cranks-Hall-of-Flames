### **文件 3: `main.js`**

```javascript
// main.js
let scene, camera, renderer;
let currentState = { x: 4, y: 4, z: 4 };
let mode = 'MOD8';
let stateSphere;

const elements = {
    0: { name: "如来藏", color: 0xffd700 },
    1: { name: "地", color: 0x8B4513 },
    2: { name: "水", color: 0x1E90FF },
    3: { name: "见", color: 0x00FF00 },
    4: { name: "空", color: 0x808080 },
    5: { name: "火", color: 0xFF4500 },
    6: { name: "风", color: 0x00FFFF },
    7: { name: "识", color: 0x9400D3 }
};

function init() {
    scene = new THREE.Scene();
    scene.background = new THREE.Color(0x050510);
    camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 0.1, 1000);
    camera.position.z = 15;
    renderer = new THREE.WebGLRenderer({ antialias: true });
    renderer.setSize(window.innerWidth, window.innerHeight);
    document.getElementById('threeContainer').appendChild(renderer.domElement);

    // Grid Helper
    const gridHelper = new THREE.GridHelper(20, 20, 0x444444, 0x222222);
    scene.add(gridHelper);

    // Axes Helper
    const axesHelper = new THREE.AxesHelper(10);
    scene.add(axesHelper);

    // Initial State Sphere
    const geometry = new THREE.SphereGeometry(0.5, 32, 32);
    const material = new THREE.MeshBasicMaterial({ color: elements[0].color });
    stateSphere = new THREE.Mesh(geometry, material);
    stateSphere.position.set(0, 0, 0);
    scene.add(stateSphere);

    updateUI();
    animate();
}

function updateSphere(state) {
    // Map Z8^3 (0-7) to Three.js coords (-3.5 to 3.5 roughly)
    const scale = 0.8;
    stateSphere.position.set(
        (state.x - 4) * scale,
        (state.y - 4) * scale,
        (state.z - 4) * scale
    );
    
    // Determine color based on dominant axis
    let color = elements[4].color; // default empty
    if (state.y >= 5) color = elements[5].color; // fire
    else if (state.y <= 3) color = elements[2].color; // water
    else if (state.x >= 5) color = elements[3].color; // seeing
    
    stateSphere.material.color.setHex(color);
}

function updateUI() {
    document.getElementById('coord').innerText = `(${currentState.x}, ${currentState.y}, ${currentState.z})`;
    let phaseName = "如来藏";
    if (currentState.y === 5) phaseName = "火大";
    if (currentState.y === 2) phaseName = "水大";
    document.getElementById('phase').innerText = phaseName;
    document.getElementById('modeText').innerText = mode;
    updateSphere(currentState);
}

// --- Event Listeners ---
document.getElementById('btnFireKing').onclick = () => {
    currentState = { x: 4, y: 5, z: 4 }; // Start at Fire
    updateUI();
    let steps = 0;
    const interval = setInterval(() => {
        let target = 3; // Seeing
        if (mode === 'XOR') {
            currentState.y = currentState.y ^ target;
        } else {
            currentState.y = (currentState.y + target) % 8;
        }
        updateUI();
        steps++;
        if (currentState.y === 0 || steps > 10) clearInterval(interval);
    }, 1000);
};

document.getElementById('btnMod8Mode').onclick = () => {
    mode = 'MOD8';
    document.querySelectorAll('.btn-toggle').forEach(btn => btn.classList.remove('active'));
    document.getElementById('btnMod8Mode').classList.add('active');
    updateUI();
};
document.getElementById('btnXorMode').onclick = () => {
    mode = 'XOR';
    document.querySelectorAll('.btn-toggle').forEach(btn => btn.classList.remove('active'));
    document.getElementById('btnXorMode').classList.add('active');
    updateUI();
};

// Background Animation
function initBackground() {
    const canvas = document.getElementById('bgCanvas');
    const ctx = canvas.getContext('2d');
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
    let particles = [];
    for (let i = 0; i < 100; i++) {
        particles.push({
            x: Math.random() * canvas.width,
            y: Math.random() * canvas.height,
            r: Math.random() * 1.5,
            d: Math.random() * 0.5
        });
    }
    function draw() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        ctx.fillStyle = "rgba(0, 245, 255, 0.3)";
        ctx.beginPath();
        for (let i = 0; i < particles.length; i++) {
            let p = particles[i];
            ctx.moveTo(p.x, p.y);
            ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2, true);
        }
        ctx.fill();
        requestAnimationFrame(draw);
    }
    draw();
}

function animate() {
    requestAnimationFrame(animate);
    renderer.render(scene, camera);
}

window.onload = () => { init(); initBackground(); };

```
